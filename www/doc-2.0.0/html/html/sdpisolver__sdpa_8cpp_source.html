1734  SCIP_SDPISOLVER* sdpisolver,
1735  SCIP_Real* lbvars,
1736  SCIP_Real* ubvars,
1737  int* arraylength
1739  )
1740 {

1742  double* X; /* block of primal solution matrix corresponding to the LP-part */
1743  int lpblockind;
1744  int nlpcons;
1745 
1746  assert( sdpisolver != NULL );
1747  assert( sdpisolver->sdpa != NULL );
1748  assert( lbvars != NULL );
1749  assert( ubvars != NULL );
1750  assert( arraylength != NULL );

1755  if ( *arraylength < sdpisolver->nvars )
1756  {
1757  *arraylength = sdpisolver->nvars;

1759  return SCIP_OKAY;
1760  }
1761 
1762  /* initialize the return-arrays with zero */
1763  for (i = 0; i < sdpisolver->nvars; i++)
1764  {
1765  lbvars[i] = 0.0;
1766  ubvars[i] = 0.0;
1767  }
1768 
1769  /* if no variable bounds were added, we return the zero vector (we do this separately, because in this case there might be no LP-block) */
1770  if ( sdpisolver->nvarbounds == 0 )
1771  {
1772  SCIPdebugMessage("Asked for PrimalBoundVars, but there were no variable bounds in sdpa, returning zero vector !");
1773  return SCIP_OKAY;
1774  }
1775 
1776  /* get the block of primal solution matrix corresponding to the LP-part from sdpa */
1777  lpblockind = (int) sdpisolver->sdpa->getBlockNumber(); /* the LP block is the last one and sdpa counts from one */
1778  assert( sdpisolver->sdpa->getBlockType((long long) lpblockind) == SDPA::LP );

1780  assert( nlpcons >= 0 );
1781 
1782  X = sdpisolver->sdpa->getResultYMat((long long) lpblockind);
1783 
1784  /* iterate over all variable bounds and insert the corresponding primal variables in the right positions of the return-arrays */
1785  assert( sdpisolver->nvarbounds <= 2 * sdpisolver->nvars || (sdpisolver->nvarbounds <= 2 * sdpisolver->nvars + 1 && sdpisolver->penalty ) );
1786  /* if we solved a penalty formulation, the last variable bound belongs to the penalty variable, which we aren't interested in here */
1787  for (i = 0; i < ((sdpisolver->penalty) ? sdpisolver->nvarbounds - 1 : sdpisolver->nvarbounds); i++)

1790  {

1792  /* the last nvarbounds entries correspond to the varbounds */
1793  lbvars[sdpisolver->sdpatoinputmapper[-1 * sdpisolver->varboundpos[i] -1]] = X[nlpcons - sdpisolver->nvarbounds + i]; /*lint !e679, !e834 */
1794  }
1795  else
1796  {
1797  /* this is an upper bound */
1798  /* the last nvarbounds entries correspond to the varbounds */
1799  ubvars[sdpisolver->sdpatoinputmapper[+1 * sdpisolver->varboundpos[i] - 1]] = X[nlpcons - sdpisolver->nvarbounds + i]; /*lint !e679, !e834 */

1801  }
1802

1804 }
