75 SCIP_RETCODE sdpRedcostFixingBinary(
76  SCIP* scip,
77  SCIP_VAR* var,
78  SCIP_Real primallbval,
79  SCIP_Real primalubval,
80  SCIP_Real cutoffbound,
81  SCIP_Real relaxval,
82  SCIP_RESULT* result
83  )
84 {
85 #ifdef SCIP_DEBUG
86  SCIP_Real sdpepsilon;
87  SCIP_Real epsilon;
88 #endif
89 
90  assert( scip != NULL );
91  assert( var != NULL );
92  assert( result != NULL );
93 
94  /* skip binary variable if it is locally fixed */
95  if (SCIPvarGetLbLocal(var) > 0.5 || SCIPvarGetUbLocal(var) < 0.5)
96  {
97  *result = SCIP_DIDNOTRUN;
98  return SCIP_OKAY;
99  }
100 
101  /* check if variable can be fixed to zero */
102  if ( SCIPisGT(scip, primallbval, cutoffbound - relaxval) )
103  {
104  SCIP_CALL( SCIPchgVarUb(scip, var, 0.0) );
105  SCIPdebugMessage("Variable %s fixed to zero by reduced cost fixing ! \n", SCIPvarGetName(var));
106 #ifdef SCIP_DEBUG
107  SCIP_CALL( SCIPgetRealParam(scip, "relaxing/SDP/sdpsolverepsilon", &sdpepsilon) );
108  SCIP_CALL( SCIPgetRealParam(scip, "numerics/epsilon", &epsilon) );
109  assert( SCIPisGE(scip, relaxval, cutoffbound) || SCIPisLE(scip, primalubval, cutoffbound - relaxval)
110  || SCIPisLE(scip, primalubval, cutoffbound - relaxval) );
111  /* if the variable should be fixed to both zero and one, something went wrong (unless we are done anyways) */
112 #endif
113  *result = SCIP_REDUCEDDOM;
114  return SCIP_OKAY;
115  }
116 
117  /* check if variable can be fixed to one */
118  if ( SCIPisGT(scip, primalubval, cutoffbound - relaxval) )
119  {
120  SCIP_CALL( SCIPchgVarLb(scip, var, 1.0) );
121  SCIPdebugMessage("Variable %s fixed to one by reduced cost fixing ! \n", SCIPvarGetName(var));
122  *result = SCIP_REDUCEDDOM;
123  return SCIP_OKAY;
124  }
125 
126  *result = SCIP_DIDNOTFIND;
127  return SCIP_OKAY;
128 }

132 SCIP_DECL_PROPEXEC(propExecSdpredcost)
133 {/*lint --e{715}*/
134  int v;
135  int nvars;
136  SCIP_VAR** vars;
137  SCIP_RELAX* relax;
138  SCIP_RESULT varresult;
139  SCIP_Real cutoffbound;
140  SCIP_Real relaxval;
141  SCIP_Bool sdpsolved;
142  SCIP_PROPDATA* propdata;
143  int length;
144 
145  SCIPdebugMessage("Calling propExecSdpredcost \n");
146 
147  assert( scip != NULL );
148  assert( prop != NULL );
149  assert( result != NULL );
150 
151  if ( SCIPgetStage(scip) == SCIP_STAGE_PRESOLVING )
152  {
153  /* we can't run before the relaxator is properly initialized */
154  *result = SCIP_DIDNOTRUN;
155  return SCIP_OKAY;
156  }
157 
158  relax = SCIPfindRelax(scip, "SDP"); /* get SDP relaxation handler */
159  assert( relax != NULL );
160 
161  /* we can only propagate for the last node for which the SDP was solved */
162  if ( SCIPrelaxSdpGetSdpNode(relax) != SCIPnodeGetNumber(SCIPgetCurrentNode(scip)) )
163  {
164  SCIPdebugMessage("Stopped propExecRedcost because current SDP-relaxation doesn't belong to the node the propagator was called for!\n");
165  *result = SCIP_DIDNOTRUN;
166  return SCIP_OKAY;
167  }
168 
169  /* we can only propagate if the SDP in the last node was solved in its original formulation */
170  if ( ! SCIPrelaxSdpSolvedOrig(relax) )
171  {
172  SCIPdebugMessage("Stopped propExecRedcost because current SDP-relaxation was solved using a penalty formulation!\n");
173  *result = SCIP_DIDNOTRUN;
174  return SCIP_OKAY;
175  }
176 
177  SCIP_CALL( SCIPrelaxSdpRelaxVal(relax, &sdpsolved, &relaxval) );
178  if ( ! sdpsolved )
179  {
180  SCIPdebugMessage("Stopped propExecRedcost because SDP-relaxation wasn't properly solved!\n");
181  *result = SCIP_DIDNOTRUN;
182  return SCIP_OKAY;
183  }
184 
185  propdata = SCIPpropGetData(prop);
186  assert( propdata != NULL );
187 
188  *result = SCIP_DIDNOTFIND;
189 
190  nvars = SCIPgetNVars(scip);
191  vars = SCIPgetVars(scip);
192 
193  cutoffbound = SCIPgetCutoffbound(scip);
194 
195  length = nvars;
196 
197  SCIP_CALL( SCIPrelaxSdpGetPrimalBoundVars(relax, propdata->lbvarvals, propdata->ubvarvals, &length) );
198 
199  assert( length == nvars ); /* we should get exactly one value for lower and upper bound-variable per variable in scip */
200 
201  for (v = 0; v < nvars; v++)
202  {
203  if ( SCIPvarIsBinary(vars[v]) )
204  {
205  SCIP_CALL( sdpRedcostFixingBinary(scip, vars[v], propdata->lbvarvals[v], propdata->ubvarvals[v], cutoffbound, relaxval, &varresult) );
206 
207  if ( varresult == SCIP_REDUCEDDOM )
208  *result = SCIP_REDUCEDDOM;
209  }
210  }
211 
212  return SCIP_OKAY;
213 }
